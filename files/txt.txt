

const db = mysql.createConnection({
  host: process.env.DB_HOST,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_DATABASE,
  port: process.env.DB_PORT
});

const currentDate = new Date().toISOString().split('T')[0];

async function checkUserSubscriptions() {
  const query = 'SELECT id, expiryDate FROM users WHERE activated = true';

  db.query(query, (err, results) => {
    if (err) {
      console.error('Error reading subscriptions from database:', err);
      return;
    }

    results.forEach(async user => {
      if (new Date(user.expiryDate) < new Date(currentDate)) {
        setTimeout(async () => {
          try {
            await bot.unbanChatMember(chatId, user.id);
            const deleteQuery = 'DELETE FROM users WHERE id = ?';
            db.query(deleteQuery, [user.id], (deleteErr) => {
              if (deleteErr) {
                console.error('Error deleting user from the database:', deleteErr);
              } else {
                console.log(`User ${user.id} removed from the database.`);
              }
            });
          } catch (error) {
            console.error(`Failed to remove user ${user.id} from the channel:`, error);
          }
        }, 10000); // الانتظار لمدة 10 ثواني
      }
    });
  });
}

let joinRequestQueue = [];
let isProcessingJoinRequests = false;

async function handleJoinRequests(request) {
  if (request) {
    joinRequestQueue.push(request);
    processJoinRequests();
  }
}

async function processJoinRequests() {
  if (isProcessingJoinRequests || joinRequestQueue.length === 0) {
    return;
  }

  isProcessingJoinRequests = true;
  const request = joinRequestQueue.shift();

  const userId = request.user_chat_id;
  const query = 'SELECT id FROM users WHERE id = ? AND activated = true';

  db.query(query, [userId], async (err, results) => {
    if (err) {
      console.error('Error reading subscriptions from database:', err);
    } else if (results.length > 0) {
      try {
        await bot.approveChatJoinRequest(chatId, userId);
        console.log(`User ${userId} approved to join the channel.`);
      } catch (error) {
        console.error(`Failed to approve user ${userId} to join the channel:`, error);
      }
    } else {
      console.log(`User ${userId} not found in active subscriptions.`);
    }

    isProcessingJoinRequests = false;
    if (joinRequestQueue.length > 0) {
      setTimeout(processJoinRequests, 10000); // الانتظار لمدة 10 ثواني قبل معالجة الطلب التالي
    }
  });
}

bot.on('chat_join_request', handleJoinRequests);

cron.schedule('0 0 * * *', () => {
  console.log('Running daily subscription check');
  checkUserSubscriptions();
});
